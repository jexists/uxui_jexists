#### 버튼 눌렀을때 글자 

```javascript
	let liLen = slideLi.length;
	for(let i=0; i<liLen; i++){ //+반복문
		slideLi.eq(i).css({zIndex:liLen-i});
	}
	let textN = 0; 
	const textEm = function(n){indiCon.text("0" + (n+1));};
	textEm(textN);

	indiBtn.on('click', function(e){
		e.preventDefault();
		if ($(this).index() == 0 ){
			textN++;
			if(textN > liLen-1){textN = 0}
			textEm(textN);
		}else{
			textN--;
			if(textN < 0){textN = liLen-1}
			textEm(textN);
		}
	})
```
